function renderChannelsb() {
  const search = document.getElementById('searchInputb').value.toLowerCase();
  const category = document.getElementById('bollywoodtv').value;
  let filtered =bollywood.filter(ch =&gt;
    ch.name.toLowerCase().includes(search) &amp;&amp;
    (category === '' || ch.group === category)
  );

  const totalPages = Math.ceil(filtered.length / perPageb);
  if (currentPageb &gt; totalPages) currentPageb = totalPages;
  if (currentPageb &lt; 1) currentPageb = 1;

  const start = (currentPageb - 1) * perPageb;
  const channels = filtered.slice(start, start + perPageb);

  document.getElementById('bollywood').innerHTML = channels.map(ch =&gt; {
    const isFav = favoritesb.includes(ch.name);
    return `
  &lt;div class=&quot;channel&quot;&gt;&lt;a href=&quot;https://www.livereacting.com/tools/mp4-player-embed?url=${ch.url}&quot; &gt;&lt;img src=&quot;${ch.logo}&quot; onerror=&quot;this.onerror=null;this.src='https://i.ibb.co/DPd27cCK/photo-2024-12-29-23-10-30.jpg';&quot;&gt;&lt;/a&gt;&lt;a href=&quot;${ch.url}&quot;&gt;
   &lt;h3&gt;${ch.name}&lt;/h3&gt;&lt;/a&gt;&lt;/div&gt;
    `;
  }).join(&quot;&quot;);

  document.getElementById('pageInfo').innerText = `Page ${currentPageb} of ${totalPages}`;
}


function toggleFav(e, name) {
  e.stopPropagation();
  if (favoritesb.includes(name)) {
    favoritesb = favoritesb.filter(f =&gt; f !== name);
  } else {
    favoritesb.push(name);
  }
  localStorage.setItem(&quot;favoritesb&quot;, JSON.stringify(favoritesb));
  renderChannelsb();
}

function nextPageb() { currentPageb++; renderChannelsb(); }
function prevPageb() { currentPageb--; renderChannelsb(); }

document.getElementById('searchInputb').addEventListener('input', renderChannelsb);
document.getElementById('bollywoodtv').addEventListener('change', renderChannelsb);

        function parseM3U(data) {
            if (!data || !data.trim().startsWith('#EXTM3U')) return [];
            const lines = data.trim().split('\n');
            const channels = [];
            for (let i = 0; i &lt; lines.length; i++) {
                if (lines[i].trim().startsWith('#EXTINF:')) {
                    let streamUrl = lines[i + 1] ? lines[i + 1].trim() : '';
                    if (!streamUrl || streamUrl.startsWith('#')) continue;
                    
                    const info = lines[i].trim();
                    const nameMatch = info.match(/tvg-name=&quot;([^&quot;]*)&quot;/) || info.match(/,(.*)$/);
                                    const idMatch = info.match(/tvg-id=&quot;([^&quot;]*)&quot;/);

                    const logoMatch = info.match(/tvg-logo=&quot;([^&quot;]*)&quot;/);
                    const groupMatch = info.match(/group-title=&quot;([^&quot;]*)&quot;/);
                    
                    channels.push({
                        name: nameMatch ? nameMatch[1].trim() : 'Unnamed',
                        logo: logoMatch ? logoMatch[1].trim() : 'https://placehold.co/280x160/1F1F1F/ffffff?text=No+Logo',
                        streamUrl: streamUrl,
                                         id: idMatch ? idMatch[1].trim() : 'no', 

                        group: groupMatch ? groupMatch[1].trim() : 'General'
                    });
                    i++;
                }
            }
            return channels;
        }

        async function loadInitialData() {
            
            try {
                const response = await fetch(playlistUrl);
                if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);
                const m3uText = await response.text();
                const allChannelsData = parseM3U(m3uText);

                if (allChannelsData.length === 0) throw new Error('No channels found.');
                
                // Logic to preserve playlist order
                const groupedChannels = {};
                const groupOrder = [];




                allChannelsData.forEach(channel =&gt; {
                    const group = channel.group || 'General';

                    
                    if (!groupedChannels[group]) {
                        groupedChannels[group] = [];
                        groupOrder.push(group); // Add group to the order array only when it's first seen
                    }
                    groupedChannels[group].push(channel);
                });

                homeCategoriesContainer.innerHTML = ''; 
                allChannelsContainer.innerHTML = '';
                
                // Render categories based on the original playlist order
